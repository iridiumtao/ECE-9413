Create `scripts/bench_ablation.sh` with the following exact content. The script (a) saves the current HEAD commit, (b) loops over the three git tags `baseline`, `exp01`, `exp03`, (c) checks out only `student.py` from each tag, (d) runs the vars20 benchmark, (e) writes everything to `bench_ablation_results.txt`, and (f) restores `student.py` to the original commit.

```bash
#!/usr/bin/env bash
set -euo pipefail

# Collect ablation benchmark data across baseline / exp01 / exp03 git tags.
# Per D-06: use git tags (not env-var flags) so each measurement is taken at the
# exact code state. Per D-07: vars20 median is the figure for the report's
# §5.4.1 ablation chart.
#
# Usage: bash scripts/bench_ablation.sh
# Output: bench_ablation_results.txt in repo root.

script_dir="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
repo_root="$(cd "${script_dir}/.." && pwd)"
cd "${repo_root}"

TAGS=(baseline exp01 exp03)
OUTFILE="${repo_root}/bench_ablation_results.txt"

# Verify all three tags exist before destructive checkout.
for tag in "${TAGS[@]}"; do
    if ! git rev-parse --verify --quiet "${tag}" >/dev/null; then
        echo "ERROR: git tag '${tag}' not found. Plan 01 Task 1 should have created baseline+exp01; Plan 01 Task 3 should have created exp03." >&2
        exit 1
    fi
done

# Save current commit so we can restore student.py after the loop.
CURRENT_COMMIT=$(git rev-parse HEAD)

# Trap restore on any exit (success, failure, Ctrl-C).
restore() {
    git checkout "${CURRENT_COMMIT}" -- student.py
    echo ""
    echo "Restore complete — student.py is back to ${CURRENT_COMMIT}."
}
trap restore EXIT

{
    echo "# Ablation benchmark — $(date)"
    echo "# Generated by scripts/bench_ablation.sh"
    echo "# Restored HEAD: ${CURRENT_COMMIT}"
    echo ""

    for tag in "${TAGS[@]}"; do
        echo "## Tag: ${tag} ($(git rev-parse "${tag}"))"
        git checkout "${tag}" -- student.py
        uv run python -m tests.benchmark --bench --bits 32 --num-vars 20 --runs 8 --warmup 3
        echo ""
    done
} | tee "${OUTFILE}"

echo ""
echo "Results saved to ${OUTFILE}"
```

```python
"""Generate the ablation bar chart for report §5.4.1.

Per D-07, shows vars20 median latency for the four base polynomials
across baseline / Exp01 / Exp03 git-tag states.

Source numbers: experiment.md (baseline + Exp01 + Exp03 sections).
Output: report/ablation_chart.png.
"""

from __future__ import annotations

from pathlib import Path

import matplotlib

matplotlib.use("Agg")  # no DISPLAY required
import matplotlib.pyplot as plt
import numpy as np


EXPRESSIONS = ["a", "a*b", "a*b + c", "a*b*c"]

# Median latency in milliseconds at num-vars 20.
BASELINE = {
    "a": 2.098,
    "a*b": 6.571,
    "a*b + c": 15.693,
    "a*b*c": 9.968,
}
EXP01 = {
    "a": 1.145,
    "a*b": 3.864,
    "a*b + c": 5.241,
    "a*b*c": 6.508,
}
EXP03 = {
    # TODO: fill from experiment.md Experiment 03 vars20 medians (Plan 02 Task 1).
    "a": <EXP03_A>,
    "a*b": <EXP03_AB>,
    "a*b + c": <EXP03_AB_PLUS_C>,
    "a*b*c": <EXP03_ABC>,
}


def main() -> None:
    x = np.arange(len(EXPRESSIONS))
    width = 0.27

    fig, ax = plt.subplots(figsize=(8.5, 5))
    bars_baseline = ax.bar(
        x - width, [BASELINE[e] for e in EXPRESSIONS], width, label="Baseline"
    )
    bars_exp01 = ax.bar(
        x, [EXP01[e] for e in EXPRESSIONS], width, label="Exp01 (t=0/t=1 shortcut)"
    )
    bars_exp03 = ax.bar(
        x + width,
        [EXP03[e] for e in EXPRESSIONS],
        width,
        label="Exp03 (filter + no-mul)",
    )

    # Annotate each bar with its numeric value above the top edge.
    for bars in (bars_baseline, bars_exp01, bars_exp03):
        for bar in bars:
            height = bar.get_height()
            ax.annotate(
                f"{height:.2f}",
                xy=(bar.get_x() + bar.get_width() / 2, height),
                xytext=(0, 3),
                textcoords="offset points",
                ha="center",
                va="bottom",
                fontsize=8,
            )

    ax.set_xlabel("Expression")
    ax.set_ylabel("Median latency (ms)")
    ax.set_title("Ablation: vars20 median latency across optimization states")
    ax.set_xticks(x)
    ax.set_xticklabels(EXPRESSIONS)
    ax.legend()
    ax.grid(axis="y", linestyle="--", alpha=0.5)

    out_path = Path("report") / "ablation_chart.png"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    fig.tight_layout()
    fig.savefig(out_path, dpi=150)
    print(f"wrote {out_path}")


if __name__ == "__main__":
    main()
```
